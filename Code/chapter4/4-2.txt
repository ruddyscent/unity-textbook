using UnityEngine;
using UnityEngine.InputSystem;

public class CarController : MonoBehaviour
{
    float speed = 0f;

    void Start()
    {
        Application.targetFrameRate = 60;
    }

    void Update()
    {
        // 스와이프의 길이를 구한다. (추가)
        Vector2 startPos = Vector2.zero;
        
        if (Mouse.current?.leftButton.wasPressedThisFrame == true ||
            Touchscreen.current?.primaryTouch.press.wasPressedThisFrame == true)
        {
            // 마우스를 클릭한 좌표
            startPos = Mouse.current != null ?
                Mouse.current.position.ReadValue() :
                Touchscreen.current.primaryTouch.position.ReadValue();
        }

        if (Mouse.current?.leftButton.wasReleasedThisFrame == true ||
            Touchscreen.current?.primaryTouch.press.wasReleasedThisFrame == true)
        {
            // 마우스를 버튼에서 손가락을 떼었을 때 좌표
            Vector2 endPos = Mouse.current != null ?
                Mouse.current.position.ReadValue() :
                Touchscreen.current.primaryTouch.position.ReadValue();

            // 스와이프 길이를 초기 속도로 변경한다.
            speed = (endPos.x - startPos.x) / 500f;
        }

        transform.Translate(this.speed, 0, 0);  // 이동
        this.speed *= 0.98f;                    // 감속
    }
}
